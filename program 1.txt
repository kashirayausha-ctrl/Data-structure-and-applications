#include<stdio.h>
#include<stdlib.h>
#define SIZE 5
struct stack{
    int top;
    int data[SIZE];
};
typedef struct stack STACK;
void push(STACK *s,int item){
    if(s->top==SIZE-1){
        printf("\nStack is full");
    }
    else{
        s->top=s->top+1;
        s->data[s->top]=item;
    }
}
void pop(STACK *s){
    if(s->top==-1){
        printf("\nStack is empty");
    }
    else{
        printf("\nElement poped is %d",s->data[s->top]);
        s->top=s->top-1;
    }
}
void display(STACK s){
    if(s.top==-1){
        printf("\nStack is empty");
    }
    else{
        printf("\nStack contents are:");
        for(int i=s.top;i>=0;i--){
            printf("%d",s.data[i]);
        }
    }
}
int main(){
    int ch,item;
    STACK s;
    s.top=-1;
    for(;;){
        printf("\n1)Push\n2)Pop\n3)Display\n4)Exit\nEnter your choice:");
        scanf("%d",&ch);
        switch(ch){
            case 1:printf("Enter number to be pushed:");
            scanf("%d",&item);
            push(&s,item);break;
            case 2:pop(&s);break;
            case 3:display(s);break;
            case 4:exit(0);
            default:printf("\nYou have entered wrong choice");
        }
    }
    return 0;
}
